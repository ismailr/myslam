118 Wall2::Wall2 (double gradient, double intercept)
119  :_gradient (gradient), _intercept (intercept), _fitness (0.0)
120 {
121  // rho = |intercept|/(gradien^2) + 1)^(0.5)

123  double rho = std::abs (_intercept) /sqrt (pow (_gradient, 2) + 1);
124  double theta = atan(-1/_gradient) * 180/M_PI;

136 void Wall2::calculate_fitness ()
137 {
138  if (_inliers.empty())
139  {
140  _fitness = 1000;
141  return;
142  }
143 
144  /* *************************
145  * Distance from a point to a line y = mx + d
146  * line eq: ax + by + c = 0
147  * a = m, b = -1, c = d
148  *
149  * distance from (x0,y0) to ax + by + c is
150  *
151  * d = |ax0 + by0 + c|
152  * ---------------
153  * sqrt (a^2 + b^2)
154  *
155  * = |mx0 - y0 + d|
156  * ---------------
157  * sqrt (m^2 + 1)
158  *
159  * ************************** */
160 
161  double d = 0.0;
162  for (Inliers::const_iterator it = _inliers.begin();
163  it != _inliers.end(); it++)
164  {
165  double x0 = (*it)[0];
166  double y0 = (*it)[1];
167  double z0 = (*it)[2];
168 
169  _fitness += std::abs (_gradient * x0 - y0 + _intercept)/sqrt (_gradient * _gradient + 1);
170  }
171 }
